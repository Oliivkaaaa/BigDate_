Задание №1.

 SELECT
 t1.name,
 sum(t2.price) AS sum_price
 FROM
 fly_mapping t3
 JOIN passengers t1 ON t1.id = t3.passenger_id
 JOIN flying t2 ON t2.id = t3.flying_id
 GROUP BY
 t1.id;

Задание №2.

 SELECT
 t1.age_group,
 round(avg(t2.flight_time), 2) AS round_flight_time
 FROM
 fly_mapping t3
 JOIN passengers t1 ON t1.id = t3.passenger_id
 JOIN flying t2 ON t2.id = t3.flying_id
 GROUP BY
 t1.age_group;

Задание №3.

 SELECT
 t1.name,
 COUNT(t2.flying_id) AS count
 FROM
 passengers t1
 JOIN fly_mapping t2 ON t1.id = t2.passenger_id
 GROUP BY
 t1.name
 HAVING
 COUNT(t2.flying_id) = (
 SELECT
 max(count)
 FROM
 (
 SELECT
 COUNT(t2.flying_id) AS count
 FROM
 fly_mapping t2
 GROUP BY
 t2.passenger_id
 )
 );

Задание №4.

 SELECT
 t1.name,
 COUNT(t2.flying_id) AS count_flying
 FROM
 passengers t1
 JOIN fly_mapping t2 ON t1.id = t2.passenger_id
 GROUP BY
 t1.name
 HAVING
 count_flying > (
 SELECT
 avg(count_flying)
 FROM
 (
 SELECT
 COUNT(t2.flying_id) AS count_flying
 FROM
 fly_mapping t2
 GROUP BY
 t2.passenger_id
 )
 );

Задание №5.

 t2.point_to,

 flying t2
 JOIN fly_mapping t3 ON t2.id = t3.flying_id
 GROUP BY
 t2.point_to
 HAVING
 count_flying = (
 SELECT
 max(count_flying)

 (
 SELECT
 COUNT(t3.flying_id) AS count_flying

 flying t2
 JOIN fly_mapping t3 ON t2.id = t3.flying_id

 t2.point_to
 )
 );
